#!/bin/sh

# Battery module for someblocks / dwmblocks
# Font Awesome 4.7 compatible

case $BLOCK_BUTTON in
    3)
        notify-send "рҹ”Ӣ Battery module" \
"пүҖ  Full
пҮҰ  Charging
пүӮ  Discharging
пҒұ  Not charging
пү„  Battery very low

Scroll: adjust brightness"
        ;;
    4) xbacklight -inc 10 ;;
    5) xbacklight -dec 10 ;;
    6) "$TERMINAL" -e "$EDITOR" "$0" ;;
esac

output=""

for battery in /sys/class/power_supply/BAT*; do
    [ -d "$battery" ] || continue

    # Space between batteries
    [ -n "$output" ] && output="$output "

    status=$(cat "$battery/status" 2>/dev/null)
    capacity=$(cat "$battery/capacity" 2>/dev/null)
    name=$(basename "$battery")

    case "$status" in
        "Full")
            icon="пүҖ"
            ;;
        "Charging")
            icon="пҮҰ"
            ;;
        "Discharging")
            if [ "$capacity" -le 15 ]; then
                icon="пү„"
            elif [ "$capacity" -le 35 ]; then
                icon="пүғ"
            elif [ "$capacity" -le 60 ]; then
                icon="пүӮ"
            elif [ "$capacity" -le 85 ]; then
                icon="пүҒ"
            else
                icon="пүҖ"
            fi
            ;;
        "Not charging")
            icon="пҒұ"
            ;;
        *)
            icon="?"
            ;;
    esac

    output="$output$name $icon ${capacity}%"
done

printf "%s\n" "$output"
