#!/bin/sh

# Prints all batteries, their percentage remaining,
# and a Font Awesome 4.7 icon corresponding to charge status.

case $BLOCK_BUTTON in
    3)
        notify-send "🔋 Battery module" \
" : discharging
 : not charging
 : charging
 : charged
 : battery very low!

- Scroll to adjust xbacklight."
        ;;
    4) xbacklight -inc 10 ;;
    5) xbacklight -dec 10 ;;
    6) "$TERMINAL" -e "$EDITOR" "$0" ;;
esac

# Loop through all attached batteries and format the info
for battery in /sys/class/power_supply/BAT*; do

    # If non-first battery, print a space separator.
    [ -n "${capacity+x}" ] && printf " "

    status_raw="$(cat "$battery/status" 2>/dev/null)"
    capacity="$(cat "$battery/capacity" 2>/dev/null)"

    case "$status_raw" in
        "Full")
            status=" "
            ;;
        "Charging")
            status=" "
            ;;
        "Discharging")
            if [ "$capacity" -le 15 ]; then
                status=" "
            elif [ "$capacity" -le 35 ]; then
                status=" "
            elif [ "$capacity" -le 60 ]; then
                status=" "
            elif [ "$capacity" -le 85 ]; then
                status=" "
            else
                status=" "
            fi
            ;;
        "Not charging")
            status=" "
            ;;
        *)
            status="? "
            ;;
    esac

    printf "%s%d%%" "$status" "$capacity"
done

printf "\n"
